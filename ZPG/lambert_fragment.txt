#version 330
in vec3 fragNormal;
in vec3 fragPos;
out vec4 frag_colour;
uniform vec3 lightPosition;
uniform vec3 lightColor;
uniform vec3 objectColor;
uniform float ambientStrength;
void main () {
    vec3 lightVector = normalize(lightPosition - fragPos);
    float dot_product = max(dot(lightVector, normalize(fragNormal)), 0.0);
    vec4 diffuse = dot_product * vec4(objectColor, 1.0);
    vec4 ambient = vec4(ambientStrength * lightColor, 1.0);
    frag_colour = (ambient + diffuse) * vec4(objectColor, 1.0);
}
